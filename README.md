## Install

```bash
$ npm install @ant-design/dark-theme
```

## Usage

```bash
$ npm install @ant-design/dark-theme --save
```

```js
import darkTheme from '@ant-design/dark-theme';

// webpack.config.js: less-loader
{
  loader: 'less-loader',
  options: {
    modifyVars: darkTheme,
  },
},
```
